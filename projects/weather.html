Weather Data Analyzer


#include <iostream>
#include <vector>
using namespace std;

int main() {
    vector<int> temps = {30, 32, 35, 28, 31};
    int sum=0, max=temps[0], min=temps[0];
    for(int t:temps) {
        sum+=t;
        if(t>max) max=t;
        if(t<min) min=t;
    }
    cout << "Avg: " << sum/temps.size() << endl;
    cout << "Max: " << max << " Min: " << min << endl;
    return 0;
}
