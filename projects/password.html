Password Generator


#include <iostream>
#include <cstdlib>
#include <ctime>
using namespace std;

int main() {
    srand(time(0));
    int length;
    cout << "Password length: ";
    cin >> length;

    string chars = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789!@#$";
    string password = "";
    for(int i=0; i<length; i++)
        password += chars[rand() % chars.size()];

    cout << "Generated Password: " << password << endl;
    return 0;
}
