Library Management


#include <iostream>
#include <map>
using namespace std;

int main() {
    map<string,bool> library;
    library["C++ Primer"] = true;
    library["DSA Book"] = true;

    string book;
    cout << "Enter book to issue: ";
    getline(cin, book);

    if(library[book]) {
        cout << "Book issued!" << endl;
        library[book] = false;
    } else {
        cout << "Not available" << endl;
    }
    return 0;
}
