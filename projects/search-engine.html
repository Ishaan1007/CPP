Mini Search Engine


#include <iostream>
#include <fstream>
#include <string>
using namespace std;

int main() {
    ifstream file("sample.txt");
    string line, word;
    cout << "Enter word to search: ";
    cin >> word;
    int lineNo=1;
    while(getline(file,line)) {
        if(line.find(word)!=string::npos)
            cout << "Found in line " << lineNo << endl;
        lineNo++;
    }
    return 0;
}
