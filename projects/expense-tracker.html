Expense Tracker


#include <iostream>
#include <fstream>
#include <string>
using namespace std;

int main() {
    ofstream file("expenses.txt", ios::app);
    string category;
    double amount;
    
    cout << "Enter category: ";
    cin >> category;
    cout << "Enter amount: ";
    cin >> amount;

    file << category << " " << amount << endl;
    file.close();
    
    cout << "Expense saved!" << endl;
    return 0;
}
