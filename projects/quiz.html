Quiz Game


#include <iostream>
using namespace std;

int main() {
    string q = "What is the size of int in C++?";
    string options[] = {"2", "4", "8"};
    int answer = 1, choice;

    cout << q << endl;
    for(int i=0;i<3;i++) cout << i+1 << ". " << options[i] << endl;
    
    cout << "Your choice: ";
    cin >> choice;
    if(choice-1 == answer) cout << "Correct!" << endl;
    else cout << "Wrong!" << endl;
    return 0;
}
